28template<typename T>
29struct Rect
30{
31 T left;
32 T top;
33 T width;
34 T height;
35
36 constexpr Rect() = default;
37 constexpr Rect(T aLeft, T aTop, T aWidth, T aHeight) : left{aLeft}, top{aTop}, width{aWidth}, height{aHeight} {}
38 constexpr Rect(const Vec2<T>& pos, const Vec2<T>& size) : left{pos.x}, top{pos.y}, width{size.x}, height{size.y} {}
39
40 constexpr bool operator==(const Rect<T>& right) const noexcept
41 {
42 return left == right.left && top == right.top && width == right.width && height == right.height;
43 }
44
45 constexpr bool operator!=(const Rect<T>& right) const noexcept
46 {

50 [[nodiscard]] constexpr bool contains(T x, T y) const noexcept
51 {
52 const T minX = std::min(left, static_cast<T>(left + width));
53 const T maxX = std::max(left, static_cast<T>(left + width));
54 const T minY = std::min(top, static_cast<T>(top + height));
55 const T maxY = std::max(top, static_cast<T>(top + height));
56
57 return (x >= minX) && (x <= maxX) && (y >= minY) && (y <= maxY);
58 }
59
60 [[nodiscard]] constexpr bool contains(const Vec2<T>& point) const noexcept
61 {
62 const T minX = std::min(left, static_cast<T>(left + width));
63 const T maxX = std::max(left, static_cast<T>(left + width));
64 const T minY = std::min(top, static_cast<T>(top + height));
65 const T maxY = std::max(top, static_cast<T>(top + height));
66
67 return (point.x >= minX) && (point.x <= maxX) && (point.y >= minY) && (point.y <= maxY);
68 }

71 {
72 return os << rect.left << ' ' << rect.top << ' ' << rect.width << ' ' << rect.height;
73 }
74};
